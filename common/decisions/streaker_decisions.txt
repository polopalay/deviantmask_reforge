	selection_tooltip = deviants_maskgo_decision_tooltip

	ai_check_interval = 12
	ai_will_do = {
		base = 0

		}

		is_in_army = no
		is_landed = yes
		exists = capital_province
		NOT = { has_trait = blind }
		NOT = { has_trait = celibate }
		NOT = { has_trait = deviants_mask }
	}
	cooldown = { years = 1 }
	effect = {
        custom_tooltip = deviants_maskgo_decision_tooltip

	}	
}
